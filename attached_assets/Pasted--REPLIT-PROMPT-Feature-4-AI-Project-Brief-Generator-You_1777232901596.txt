**Request body:**
```json

}
```

**Anthropic API call:**

Headers:
```
x-api-key: process.env.ANTHROPIC_API_KEY
anthropic-version: 2023-06-01
content-type: application/json
```

Body:
```json
{
  "model": "claude-sonnet-4-20250514",
  "max_tokens": 1500,
  "messages": [
    {
      "role": "user",
      "content": "You are a senior professional services delivery manager. Based on the engagement description below, generate a structured project plan in JSON format only.     }
  ]

```

Replace `{description}`, `{roles}`, and `{templateExamples}` with actual values before sending.

If `rateCardId` is not provided or rate card fetch fails, set `{roles}` to `"Solution Architect, Functional Consultant, Technical Consultant, Project Manager, Business Analyst"`.

**Response handling:**

- Validate: the result must be a valid JSON object with `phases` as a non-empty array
- If parsing fails or Anthropic does not respond within 10 seconds, return HTTP 422:
```json
  { "error": "generation_failed", "message": "Could not generate a project brief. Please try again or start from a template." }
```
- On success, return the parsed JSON object directly with HTTP 200

Register this route in `artifacts/api-server/src/routes/index.ts`. Do not remove or change any existing route registrations.

**On "Generate Project Plan" click:**

```typescript
async function handleGenerate() {
  setGeneratorLoading(true);
  setGeneratorError(null);
  setGeneratedBrief(null);
  try {
    const res = await fetch("/api/ai/generate-project-brief", {
      method: "POST",
      headers: { "Content-Type": "application/json", ...authHeaders() },
      body: JSON.stringify({
        description: generatorDescription,
        rateCardId: generatorRateCardId,
      }),
    });
    if (!res.ok) {
      const err = await res.json().catch(() => ({}));
      throw new Error(err.message ?? "Generation failed. Please try again.");
    }
    const brief: GeneratedBrief = await res.json();
    setGeneratedBrief(brief);
  } catch (e: any) {
    setGeneratorError(e.message ?? "Something went wrong. Please try again.");
  } finally {
    setGeneratorLoading(false);
  }
}
```
